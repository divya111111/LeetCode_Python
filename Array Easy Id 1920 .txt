1920. Build Array from Permutation


Sol 1.


class Solution(object):
    def buildArray(self, nums):
        index=nums
        new_arr=[]                                                  
        for i in range(0, len(nums)):
            arr=nums
            index=nums
            new_arr.append(arr[index[i]])
        
        return new_arr
